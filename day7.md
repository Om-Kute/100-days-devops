# Day 7 - Archiving & File Utilities

tar -cvf archive.tar file.txt   # Create archive
tar -xvf archive.tar            # Extract archive
basename /path/file.txt         # Get filename
compress file.txt               # Compress file (old)
cpio                           # Archive files
csplit file.txt                # Split file
expand file.txt                # Tabs to spaces
file file.txt                  # File type
fold file.txt                  # Wrap text
less file.txt                  # View file page by page
