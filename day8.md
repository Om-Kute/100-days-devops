# Day 8 - Directory & File Operations

join file1 file2        # Join files based on field
split file.txt          # Split file into parts
unexpand file.txt       # Convert spaces to tabs
dir                     # List directory
dirs                    # Show directory stack
dirname /path/file.txt  # Get directory path
lsblk                   # Show disk partitions
mount                   # Mount file system
tree                    # Directory tree view
du -sh folder/          # Disk usage of folder
